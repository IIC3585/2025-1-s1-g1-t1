(constant) toHtmlTable

Description:

  • Convert an array of arrays into an HTML table.

Source:
  • parser_html.js, line 64

Convert an array of arrays into an HTML table.

Example
const file = [
  ["a", "b", "c"],
  ["d", "e", "f"],
  ["g", "h", "i"]
];
const htmlTable = toHtmlTable(file);
console.log(htmlTable);
// Output:
// <table>
//     <tr>
//         <td>a</td>
//         <td>b</td>
//         <td>c</td>
//     </tr>
//     <tr>
//         <td>d</td>
//         <td>e</td>
//         <td>f</td>
//     </tr>
//     <tr>
//         <td>g</td>
//         <td>h</td>
//         <td>i</td>

insertColumn(file, n, column) → {Array.<Array.<any>>}

Description:

  • This function inserts a column into a file at the specified position. It can insert a column at the head (0), tail (-1), or any other position (n). The column must match the number of rows in the file.

Source:
  • insertion.js, line 40
Examples
const file = [
  ["a", "b", "c"],
  ["d", "e", "f"],
  ["g", "h", "i"]
];
const column = ["1", "2", "3"];
const n = 1;
const newFile = insertColumn(file, n, column);
// console.log(newFile);
// Output:
// [
//   ["a", "1", "b", "c"],
//   ["d", "2", "e", "f"],
//   ["g", "3", "h", "i"]
// ];
const column = ["1", "2", "3"];
const n = -1;
const newFile = insertColumn(file, n, column);
// console.log(newFile);
// Output:
// [
//   ["a", "b", "c", "1"],
//   ["d", "e", "f", "2"],
//   ["g", "h", "i", "3"]
// ];

transpose(matrix) → {Array}

Description:

  • Transposes a matrix, converting rows into columns and vice versa. It is used for both requested functions.

Source:
  • mutation.js, line 6
Example
const file = [
  ["a", "b", "c"],
  ["d", "e", "f"],
  ["g", "h", "i"]
];
const rowsToColumns = transpose(file);
const columnsToRows = transpose(rowsToColumns);
console.log(rowsToColumns);

// Output: [
 ["a", "d", "g"],
 ["b", "e", "h"],
 ["c", "f", "i"]
];

console.log(columnsToRows);

// Output: [
["a", "b", "c"],
["d", "e", "f"],
["g", "h", "i"]
];
